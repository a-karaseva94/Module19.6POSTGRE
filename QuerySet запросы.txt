# запрос с агрегацией (максимальная, минимальная и средняя цены игр)

from django.db.models import Max, Min, Avg
stats = Games.objects.aggregate(
max_price=Max('cost'),
min_price=Min('cost'),
avg_price=Avg('cost')
)
print(stats)


#запрос с фильтрацией - игры дешевле 11000
